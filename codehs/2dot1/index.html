function start() {
    for (var row = 0; row < 5; row++) {
        cleanRow();
        if (row < 4) moveToNextRow();
    }
    moveToFarRight();
}
function cleanRow() {
    while (frontIsClear()) {
        if (colorIs(Color.purple)||colorIs(Color.green)) paint(Color.yellow);
        else paint(Color.gray);
        move();
    }
}
function moveToNextRow() { turnLeft(); move(); turnLeft(); }
function moveToFarRight() { while(frontIsClear()) move(); }
